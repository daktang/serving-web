N = int(input())

B = [[-1] * N for _ in range(N)]

# 삼각형 영역만 사용하기 위한 준비
for i in range(N):
    for j in range(i + 1):
        B[i][j] = -1

print(B)

dr = [1, 0, -1]
dc = [1, -1, 0]

r, c = 0, 0
direction = 0

# 총 값의 개수: 1, 2, 3, ..., N = N * (N + 1) // 2
total = N * (N + 1) // 2
num = 0

for _ in range(total):
    B[r][c] = num % 10
    num += 1

    print(num)
    print(B)

    nr = r + dr[direction]
    nc = c + dc[direction]

    if nr < 0 or nc < 0 or nr >= N or nc >= N or nc > nr or B[nr][nc] != -1:
        direction = (direction + 1) % 3

        nr = r + dr[direction]
        nc = c + dc[direction]

    r, c = nr, nr

for i in range(N):
    for j in range(i + 1):
        print(B[i][j], end=" ")
    print()
